Phase 1: Seed URL & robots.txt Checking

The crawl cycle begins by submitting a seed URL. Before any fetching tasks are scheduled, the orchestrator requests the domain's /robots.txt file and parses its contents. This file defines directory accessibility rules for automated agents. The orchestrator checks if path targets are allowed by user-agent rules (like Disallow: /admin/) before adding links to the active crawl queue, guaranteeing safety and compliance.

Python code equivalent: 
rp = urllib.robotparser.RobotFileParser()
rp.set_url(seed_url + "/robots.txt")
rp.read()
is_allowed = rp.can_fetch("*", target_url)
# crawler.py - Standalone Asynchronous High-Performance Web Crawler import asyncio import aiohttp from urllib.parse import urlparse, urljoin from bs4 import BeautifulSoup import sqlite3 class AsyncWebCrawler: def __init__(self, seed_url, max_depth=3, max_pages=50, concurrency=5): self.seed_url = seed_url self.target_domain = urlparse(seed_url).netloc self.max_depth = max_depth self.max_pages = max_pages self.semaphore = asyncio.Semaphore(concurrency) self.visited_urls = set() self.queue = asyncio.Queue() async def fetch_and_parse(self, url, depth): async with self.semaphore: try: # Polite concurrency requests & header parsing async with self.session.get(url, timeout=10) as response: html = await response.text() self.visited_urls.add(url) soup = BeautifulSoup(html, 'html.parser') title = soup.title.string if soup.title else url # SQLite Indexing self.save_database(url, title, response.status) if depth < self.max_depth: for link in soup.find_all('a', href=True): abs_link = urljoin(url, link['href']) if urlparse(abs_link).netloc == self.target_domain: await self.queue.put((abs_link, depth + 1)) except Exception as e: print(f"Fetch failure {url}: {e}")
